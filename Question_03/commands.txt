#!/bin/bash
fail1=0; passAll=0
while IFS=',' read -r roll name m1 m2 m3; do
    fail_count=0
    (( m1 < 33 )) && ((fail_count++))
    (( m2 < 33 )) && ((fail_count++))
    (( m3 < 33 )) && ((fail_count++))
    
    if [ $fail_count -eq 1 ]; then
        echo "Failed in 1: $name"; ((fail1++))
    elif [ $fail_count -eq 0 ]; then
        echo "Passed All: $name"; ((passAll++))
    fi
done < marks.txt
echo "Summary: 1 Fail: $fail1, Passed All: $passAll"
